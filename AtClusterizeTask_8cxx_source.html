94 void AtClusterizeTask::processPoint(Int_t mcPointID)
95 {
96  auto trackID = fMCPoint->GetTrackID();
97 
98  // If it is a new track entering the volume or no energy was deposited
99  // record its location and the new track ID
100  if (fMCPoint->GetEnergyLoss() == 0 || fCurrTrackID != trackID) {
101  setNewTrack();
102  std::cout << "Point is new track" << std::endl;
103  return;
104  }
105 
106  XYZVector currentPoint = getCurrentPointLocation(); // [mm, mm, us]
107  auto genElectrons = getNumberOfElectronsGenerated();
108 
109  XYZVector step;
110  if (genElectrons > 0) {
111  step = (currentPoint - fPrevPoint) / genElectrons;
112  }
113 
114  auto sigTrans = getTransverseDiffusion(currentPoint.z()); // mm
115  auto sigLong = getLongitudinalDiffusion(currentPoint.z()); // us
116 
117  // Need to loop through electrons
118  for (int i = 0; i < genElectrons; ++i) {
119  auto loc = applyDiffusion(currentPoint + i * step, sigTrans, sigLong);
120  auto size = fSimulatedPointArray->GetEntriesFast();
121  new ((*fSimulatedPointArray)[size]) AtSimulatedPoint(mcPointID, i, loc);
122  }
123 
124  fPrevPoint = currentPoint;
125 }

171 AtClusterizeTask::applyDiffusion(const ROOT::Math::XYZVector &loc, double_t sigTrans, double sigLong)
172 {
173  auto r = gRandom->Gaus(0, sigTrans);

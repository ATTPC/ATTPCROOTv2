109  Myy += Yi * Yi;
110  Mxz += Xi * Zi;
111  Myz += Yi * Zi;
112  Mzz += Zi * Zi;
113  }

117  Mxy /= points.size();

131  A22 = A2 + A2;
132  A33 = A3 + A3 + A3;

136  // (it is guaranteed to converge to the right root)

142  if ((xnew == x) || (!std::isfinite(xnew)))

156  acos(((from->at(0)) * (to->at(0)) + (from->at(1)) * (to->at(1)) + (from->at(2)) * (to->at(2))) / (normf * normt));

158  b = 1.0 - cos(alpha);
159

167  n[1] = ((from->at(2)) * (to->at(0)) - (from->at(0)) * (to->at(2)));
168  n[2] = ((from->at(0)) * (to->at(1)) - (from->at(1)) * (to->at(0)));

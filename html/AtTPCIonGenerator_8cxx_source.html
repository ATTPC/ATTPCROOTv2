136 // ----- Private method SetEmittance ----------------------------------
137 void AtTPCIonGenerator::SetEmittance()
138 {
139 
140  Double_t x = 0., y = 0., xFocus = 0., yFocus = 0., theta = 0., phi = 0.;
141  Double_t ptot = sqrt(pow(fPx, 2) + pow(fPy, 2) + pow(fPz, 2));
142 
143  /*
144  Ex.
145  fWhmFocus = 1.; //cm, FWHM of Gaussian
146  fDiv = 10.*1E-3; //radians
147  fZFocus = 50; //cm, focus distance from entrance
148  */
149  // x is coordinates of beam particle at AtTPC entrance, xFocus is coordinates at focus.
150  xFocus = gRandom->Gaus(0, fWhmFocus / 2.355);
151  yFocus = gRandom->Gaus(0, fWhmFocus / 2.355);
152 
153  do {
154  theta = gRandom->Uniform(-fDiv, fDiv);
155  phi = gRandom->Uniform(-fDiv, fDiv);
156  x = xFocus - fZFocus * tan(phi);
157  y = yFocus - sqrt(pow(fZFocus, 2) + pow(xFocus - x, 2)) * tan(theta);
158  } while (sqrt(pow(x, 2) + pow(y, 2)) > fRHole && sqrt(pow(tan(theta), 2) + pow(tan(phi), 2)) > tan(fDiv));
159 
160  fVx = x;
161  fVy = y;
162  fVz = 0.;
163 
164  fPx = ptot * cos(theta) * sin(phi);
165  fPy = ptot * sin(theta);
166  fPz = sqrt(ptot * ptot - fPx * fPx - fPy * fPy);
167 
168  AtVertexPropagator::Instance()->Setd2HeVtx(fVx, fVy, theta, phi);
169 }

83 bool AtFilterFFT::doesFreqRangeOverlap(const AtFreqRange &newRange)
84 {
85  for (auto &range : fFreqRanges) {
86  // Check to make sure the minimum frequency does not exist within the bounds
87  // of any added frequency, except for the max frequency when the factors are
88  // the same
89  if (newRange.fBeginFreq >= range.fBeginFreq && newRange.fBeginFreq < range.fEndFreq)
90  return true;
91  if (newRange.fBeginFreq == range.fEndFreq && newRange.fBeginFact != range.fEndFact)
92  return true;
93 
94  // Check to make sure the maximum frequency does not exist within the bounds of any added
95  // frequency, except for when it matches the minimum frequency with the same factor
96  if (newRange.fEndFreq > range.fBeginFreq && newRange.fEndFreq <= range.fEndFreq)
97  return true;
98  if (newRange.fEndFreq == range.fBeginFreq && newRange.fEndFact != range.fBeginFreq)
99  return true;
100  }
101 
102  return false;

373 void AtRANSACN::AtRansac::FindVertex(std::vector<AtTrack *> tracks)
374 {
375 
376  // Assumes the minimum distance between two lines, with respect a given threshold, the first vertex candidate. Then
377  // evaluates the distance of each remaining line with respect to the others (vertex) to decide the particles of the
378  // reaction.
379  // std::cout<<" New find vertex call "<<std::endl;
380 
381  Double_t mad = 10; // Minimum approach distance. This is the minimum distance between the lines in 3D. Must be bigger
382  // than fLineDistThreshold
383  XYZVector c_1(-1000, -1000, -1000);
384  XYZVector c_2(-1000, -1000, -1000);
385  // std::vector<AtTrack*> *TrackCand;
386 
387  // Current parametrization
388  // x = p[0] + p[1]*t;
389  // y = p[2] + p[3]*t;
390  // z = t;
391  // (x,y,z) = (p[0],p[2],0) + (p[1],p[3],1)*t
392 
393  // Equation of the Z axis in the solenoid frame
394  XYZVector Z_0(0, 0, 1000.0);
395  XYZVector Z_1(0, 0, 1);
396 
397  // Equation of the Y axis in the
398  XYZVector Y_0(0, 0, 1000.0);
399  XYZVector Y_1(0, 1, 0);
400 
401  // Counter clockwise rotations
402  // Direction of the Z and Y axis in the detector frame (This is the rotated detector), this is used to calculater the
403  // theta and phi angles
404  RotationX rx(-fTiltAng * TMath::Pi() / 180.0);
405  // XYZVector Z_1_rot(rx * Z_1);
406 
407  RotationX ry(-fTiltAng * TMath::Pi() / 180.0);
408  // XYZVector Y_1_rot(ry * Y_1);
409 
410  // Vector of the beam determined from the experimental data
411  // XYZVector BeamDir_1(-0.106359,-0.0348344,1.0);
412  XYZVector BeamDir_1(0., 0., 1.0);
413  // TODO:: This is for 6.5 degrees of tilting angle. Need a function to set it.
414 
415  // Test each line against the others to find a vertex candidate
416  for (Int_t i = 0; i < int(tracks.size()) - 1; i++) {
417 
418  AtTrack *track = tracks.at(i);
419  track->SetTrackID(i);
420  std::vector<Double_t> p = track->GetFitPar();
421 
422  if (p.size() > 0) {
423  // XYZVector L_0(p[0], p[2], 0. );//p1
424  // XYZVector L_1(p[1], p[3], 1. );//d1
425  XYZVector L_0(p[0], p[2], p[4]); // p1
426  XYZVector L_1(p[1], p[3], p[5]); // d1
427 
428  // std::cout<<" L_1 p[0] : "<<p[0]<<" L_1 p[2] : "<<p[2]<<std::endl;
429  // std::cout<<" L_1 p[1] : "<<p[1]<<" L_1 p[3] : "<<p[3]<<std::endl;
430 
431  for (Int_t j = i + 1; j < tracks.size(); j++) {
432  AtTrack *track_f = tracks.at(j);
433  track_f->SetTrackID(j);
434  std::vector<Double_t> p_f = track_f->GetFitPar();
435 
436  if (p_f.size() > 0) {
437  // XYZVector L_f0(p_f[0], p_f[2], 0. );//p2
438  // XYZVector L_f1(p_f[1], p_f[3], 1. );//d2
439  XYZVector L_f0(p_f[0], p_f[2], p_f[4]); // p2
440  XYZVector L_f1(p_f[1], p_f[3], p_f[5]); // d2
441 
442  // std::cout<<" L_f0 p_f[0] : "<<p_f[0]<<" L_f1 p_f[2] : "<<p_f[2]<<std::endl;
443  // std::cout<<" L_f1 p_f[1] : "<<p_f[1]<<" L_f1 p_f[3] : "<<p_f[3]<<std::endl;
444 
445  XYZVector L = L_1.Cross(L_f1);
446  Double_t L_mag = L.Rho();
447  XYZVector n_1 = L_1.Cross(L);
448  XYZVector n_2 = L_f1.Cross(L);
449  c_1 = L_0 + ((L_f0 - L_0).Dot(n_2) * L_1) / (L_1.Dot(n_2));
450  c_2 = L_f0 + ((L_0 - L_f0).Dot(n_1) * L_f1) / (L_f1.Dot(n_1));
451 
452  // std::cout<<i<<" "<<j<<" "<<L_mag<<std::endl;
453 
454  if (L_mag > 0) {
455  XYZVector n = L / (Double_t)L_mag;
456  Double_t d = TMath::Abs(n.Dot(L_0 - L_f0));
457  // std::cout<<" Distance of minimum approach : "<<d<<std::endl;
458  Double_t num = L_1.X() * L_f1.X() + L_1.Y() * L_f1.Y() + L_1.Z() * L_f1.Z();
459  Double_t den = TMath::Sqrt(L_1.X() * L_1.X() + L_1.Y() * L_1.Y() + L_1.Z() * L_1.Z()) *
460  TMath::Sqrt(L_f1.X() * L_f1.X() + L_f1.Y() * L_f1.Y() + L_f1.Z() * L_f1.Z());
461  // NB:: The vector of hits is always sorted in time so the direction of porpagation of the particle is
462  // always positive. This means that the angle will be always between 0 and 90 degree. The vertex and
463  // the mean time of the track are needed to determine the direction of the track and then add 90
464  // degrees.
465  Double_t ang2 = TMath::ACos(num / den);
466  XYZVector vertex1_buff;
467  vertex1_buff.SetXYZ(c_1.X(), c_1.Y(), c_1.Z());
468  XYZVector vertex2_buff;
469  vertex2_buff.SetXYZ(c_2.X(), c_2.Y(), c_2.Z());
470 
471  // Angle with respect to solenoid
472  Double_t angZi = GetAngleTracks(L_1, Z_1);
473  Double_t angZj = GetAngleTracks(L_f1, Z_1);
474 
475  // Angles with respect to tilted detector
476  Double_t angZDeti = GetAngleTracks(L_1, BeamDir_1);
477  Double_t angZDetj = GetAngleTracks(L_f1, BeamDir_1);
478  Double_t angYDeti = 0.0; // GetAngleTracks(L_1,Y_1_rot);
479  Double_t angYDetj = 0.0; // GetAngleTracks(L_f1,Y_1_rot);
480 
481  track->SetAngleZAxis(angZi);
482  track->SetAngleZDet(angZDeti);
483  track->SetAngleYDet(angYDeti);
484 
485  track_f->SetAngleZAxis(angZj);
486  track_f->SetAngleZDet(angZDetj);
487  track_f->SetAngleYDet(angYDetj);
488 
489  XYZPoint vertexAvg(0.5 * (vertex1_buff + vertex2_buff));
490  track->SetTrackVertex(vertexAvg);
491  track_f->SetTrackVertex(vertexAvg);
492 
493  if (d < mad) {
494 
495  mad = d;
496  // std::cout<<" New distance of minimum approach : "<<mad<<std::endl;
497  // std::cout<<" Angle between lines i : "<<i<<" j : "<<j<<" "<<ang2<<std::endl;
498 
499  // Global event variables
500  fVertex_1.SetXYZ(c_1.X(), c_1.Y(), c_1.Z());
501  fVertex_2.SetXYZ(c_2.X(), c_2.Y(), c_2.Z());
502  fVertex_mean = 0.5 * (fVertex_1 + fVertex_2);
503  fVertex_tracks.first = i;
504  fVertex_tracks.second = j;
505  fMinimum = mad;
506 
507  if (!CheckTrackID(track->GetTrackID(), &fTrackCand)) {
508  fTrackCand.push_back(*track);
509  PairedLines PL;
510  PL.LinesID.first = i;
511  PL.LinesID.second = j;
512  PL.AngleZAxis.first = angZi;
513  PL.AngleZAxis.second = angZj;
514  PL.AngleZDet.first = angZDeti;
515  PL.AngleZDet.second = angZDetj;
516  PL.AngleYDet.first = angYDeti;
517  PL.AngleYDet.second = angYDetj;
518  PL.minDist = mad;
519  PL.meanVertex = fVertex_mean;
520  PL.angle = ang2;
521  PLines.push_back(PL);
522  }
523  if (!CheckTrackID(track_f->GetTrackID(), &fTrackCand)) {
524  fTrackCand.push_back(*track_f);
525  PairedLines PL;
526  PL.LinesID.first = i;
527  PL.LinesID.second = j;
528  PL.AngleZAxis.first = angZi;
529  PL.AngleZAxis.second = angZj;
530  PL.AngleZDet.first = angZDeti;
531  PL.AngleZDet.second = angZDetj;
532  PL.AngleYDet.first = angYDeti;
533  PL.AngleYDet.second = angYDetj;
534  PL.minDist = mad;
535  PL.meanVertex = fVertex_mean;
536  PL.angle = ang2;
537  PLines.push_back(PL);
538  }
539  }
540 
541  if (d < fLineDistThreshold) {

591 Double_t AtRANSACN::AtRansac::distance2(double x, double y, double z, const double *p)
592 {
593 
594  // distance line point is D= | (xp-x0) cross ux |
595  // where ux is direction of line and x0 is a point in the line (like t = 0) and x1 is in t=1
596  XYZVector xp(x, y, z);
597  XYZVector x0(p[0], p[2], 0.);
598  XYZVector x1(p[0] + p[1], p[2] + p[3], 1.);
599  XYZVector u = (x1 - x0).Unit();
600  double d2 = ((xp - x0).Cross(u)).Mag2();
601  return d2;
602 }

621 Double_t AtRANSACN::AtRansac::GetAngleTracks(const ROOT::Math::XYZVector &vec1, const ROOT::Math::XYZVector &vec2)
622 {
623 
624  // NB:: The vector of hits is always sorted in time so the direction of porpagation of the particle is always
625  // positive. This means that the angle will be always between 0 and 90 degree. The vertex and the mean time of the
626  // track are needed to determine the direction of the track and then add 90 degrees.
627  Double_t num = vec1.X() * vec2.X() + vec1.Y() * vec2.Y() + vec1.Z() * vec2.Z();
628  Double_t den = TMath::Sqrt(vec1.X() * vec1.X() + vec1.Y() * vec1.Y() + vec1.Z() * vec1.Z()) *
629  TMath::Sqrt(vec2.X() * vec2.X() + vec2.Y() * vec2.Y() + vec2.Z() * vec2.Z());
630  Double_t ang = TMath::ACos(num / den);
631 
632  return ang;

646 Double_t AtRANSACN::AtRansac::Fit3D(AtTrack *track)
647 {
648 
649  //--------Adapted from https://www.scribd.com/doc/31477970/Regressions-et-trajectoires-3D
650  //--------3D linear regression
651 
652  int R, C;
653  double Q;
654  double Xm, Ym, Zm;
655  double Xh, Yh, Zh;
656  double a, b;
657  double Sxx, Sxy, Syy, Sxz, Szz, Syz;
658  double theta;
659  double K11, K22, K12, K10, K01, K00;
660  double c0, c1, c2;
661  double p, q, r, dm2;
662  double rho, phi;
663 
664  Q = Xm = Ym = Zm = 0.;
665  Sxx = Syy = Szz = Sxy = Sxz = Syz = 0.;
666 
667  std::vector<AtHit> HitArray = track->GetHitArray();
668  std::vector<double> X;
669  std::vector<double> Y;
670  std::vector<double> Z;
671  std::vector<double> Charge;
672 
673  for (auto hit : HitArray) {
674  auto pos = hit.GetPosition();
675  // Double_t tq = hit.GetCharge();
676  X.push_back(pos.X());
677  Y.push_back(pos.Y());
678  Z.push_back(pos.Z());
679  Charge.push_back(q);
680  }
681 
682  for (Int_t i = 0; i < HitArray.size(); i++) {
683  Q += Charge[i] / 10.;
684  Xm += X[i] * Charge[i] / 10.;
685  Ym += Y[i] * Charge[i] / 10.;
686  Zm += Z[i] * Charge[i] / 10.;
687  Sxx += X[i] * X[i] * Charge[i] / 10.;
688  Syy += Y[i] * Y[i] * Charge[i] / 10.;
689  Szz += Z[i] * Z[i] * Charge[i] / 10.;
690  Sxy += X[i] * Y[i] * Charge[i] / 10.;
691  Sxz += X[i] * Z[i] * Charge[i] / 10.;
692  Syz += Y[i] * Z[i] * Charge[i] / 10.;
693  }
694 
695  Xm /= Q;
696  Ym /= Q;
697  Zm /= Q;
698  Sxx /= Q;
699  Syy /= Q;
700  Szz /= Q;
701  Sxy /= Q;
702  Sxz /= Q;
703  Syz /= Q;
704  Sxx -= (Xm * Xm);
705  Syy -= (Ym * Ym);
706  Szz -= (Zm * Zm);
707  Sxy -= (Xm * Ym);
708  Sxz -= (Xm * Zm);
709  Syz -= (Ym * Zm);
710 
711  theta = 0.5 * atan((2. * Sxy) / (Sxx - Syy));
712 
713  K11 = (Syy + Szz) * pow(cos(theta), 2) + (Sxx + Szz) * pow(sin(theta), 2) - 2. * Sxy * cos(theta) * sin(theta);
714  K22 = (Syy + Szz) * pow(sin(theta), 2) + (Sxx + Szz) * pow(cos(theta), 2) + 2. * Sxy * cos(theta) * sin(theta);
715  // K12 = -Sxy * (pow(cos(theta), 2) - pow(sin(theta), 2)) + (Sxx - Syy) * cos(theta) * sin(theta);
716  K10 = Sxz * cos(theta) + Syz * sin(theta);
717  K01 = -Sxz * sin(theta) + Syz * cos(theta);
718  K00 = Sxx + Syy;
719 
720  c2 = -K00 - K11 - K22;
721  c1 = K00 * K11 + K00 * K22 + K11 * K22 - K01 * K01 - K10 * K10;
722  c0 = K01 * K01 * K11 + K10 * K10 * K22 - K00 * K11 * K22;
723 
724  p = c1 - pow(c2, 2) / 3.;
725  q = 2. * pow(c2, 3) / 27. - c1 * c2 / 3. + c0;
726  r = pow(q / 2., 2) + pow(p, 3) / 27.;
727 
728  if (r > 0)
729  dm2 = -c2 / 3. + pow(-q / 2. + sqrt(r), 1. / 3.) + pow(-q / 2. - sqrt(r), 1. / 3.);
730  else {
731  rho = sqrt(-pow(p, 3) / 27.);
732  phi = acos(-q / (2. * rho));
733  dm2 = std::min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos(phi / 3.),
734  std::min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 2. * TMath::Pi()) / 3.),
735  -c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 4. * TMath::Pi()) / 3.)));
736  }
737 
738  a = -K10 * cos(theta) / (K11 - dm2) + K01 * sin(theta) / (K22 - dm2);
739  b = -K10 * sin(theta) / (K11 - dm2) - K01 * cos(theta) / (K22 - dm2);
740 
741  Xh = ((1. + b * b) * Xm - a * b * Ym + a * Zm) / (1. + a * a + b * b);
742  Yh = ((1. + a * a) * Ym - a * b * Xm + b * Zm) / (1. + a * a + b * b);
743  Zh = ((a * a + b * b) * Zm + a * Xm + b * Ym) / (1. + a * a + b * b);
744 
745  std::vector<Double_t> par;
746  par.push_back(Xm); // 0
747  par.push_back(Xh - Xm); // 1
748  par.push_back(Ym); // 2
749  par.push_back(Yh - Ym); // 3
750  par.push_back(Zm); // 4
751  par.push_back(Zh - Zm); // 5
752 
753  track->SetFitPar(par);
754  track->SetMinimum(fabs(dm2 / Q));
755  track->SetNFree(X.size() - 6);
756  X.clear();
757  Y.clear();
758  Z.clear();
759  Charge.clear();
760 
761  return (fabs(dm2 / Q));
762 }

211

213  double ran1 = (gRandom->Uniform());
214  double ran2 = (gRandom->Uniform());
215  double thetapn = acos(2 * ran1 - 1.);
216  double phipn = 2 * TMath::Pi() * ran2;
217 
218  // generate the p and n in the 2H rest frame
219  double pprest[3], pnrest[3];
220  pprest[2] = Pcpn * cos(thetapn);
221  pprest[0] = Pcpn * sin(thetapn) * cos(phipn);
222  pprest[1] = Pcpn * sin(thetapn) * sin(phipn);
223  pnrest[2] = -1 * pprest[2];
224  pnrest[0] = -1 * pprest[0];
225  pnrest[1] = -1 * pprest[1];
226  double Eprest = sqrt(pow(Pcpn, 2) + pow(mp, 2));
227  double Enrest = sqrt(pow(Pcpn, 2) + pow(mn, 2));
228 
229  // boost to 2He frame
230  double ppL[3], pnL[3];
231  ppL[0] = pprest[0];
232  ppL[1] = pprest[1];
233  ppL[2] = gammad * (pprest[2] + betad * Eprest);
234  // double EpL = sqrt(pow(mp, 2) + pow(ppL[0], 2) + pow(ppL[1], 2) + pow(ppL[2], 2));
235

237  pnL[1] = pnrest[1];
238  pnL[2] = gammad * (pnrest[2] + betad * Enrest);
239  // double EnL = sqrt(pow(mn, 2) + pow(pnL[0], 2) + pow(pnL[1], 2) + pow(pnL[2], 2));
240 
241  // rotate to the 2H direction

243  std::vector<double> fvto(3);
244  std::vector<double> fvin(3);
245  std::vector<double> fvout(3);
246  fvfrom.at(0) = 0;
247  fvfrom.at(1) = 0;
248  fvfrom.at(2) = 1;

251  fvto.at(2) = pdL[2];

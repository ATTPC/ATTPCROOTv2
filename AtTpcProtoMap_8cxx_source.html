155 Bool_t AtTpcProtoMap::SetProtoMap(TString file)
156 {
157 
158  /* Sets a map for the prototype containing the Pad Number*/
159  TString PadName;
160  Int_t PadNum = -1;
161  Float_t pad_xcoord = -1;
162  Float_t pad_ycoord = -1;
163  Int_t bin_num = -1;
164  std::vector<Float_t> PadCoord;
165  PadCoord.reserve(2);
166  auto InProtoMap = std::make_unique<std::ifstream>(file.Data());
167 
168  if (InProtoMap->fail()) {
169  std::cout << " = AtTpcProtoMap::SetProtoMap : No Prototype Map file found! Please, check the path. Current :"
170  << file.Data() << std::endl;
171  return -1;
172  } else
173  kIsProtoMapSet = kTRUE;
174 
175  while (!InProtoMap->eof()) {
176 
177  *InProtoMap >> PadNum >> PadName >> pad_xcoord >> pad_ycoord >> bin_num;
178  PadCoord.push_back(pad_xcoord);
179  PadCoord.push_back(pad_ycoord);
180  if (kDebug)
181  if (!PadName.IsNull())
182  std::cout << " PadNum : " << PadNum << " - PadName : " << PadName.Data()
183  << " - Pad X coord : " << pad_xcoord << " - Pad Y coord : " << pad_ycoord
184  << " - Bin number : " << bin_num << std::endl;
185  ProtoGeoMap.insert(std::pair<Int_t, std::vector<Float_t>>(PadNum, PadCoord));
186  ProtoBinMap.insert(std::pair<Int_t, Int_t>(bin_num, PadNum));
187  PadCoord.clear();
188  }
189 
190  return kTRUE;
191 }

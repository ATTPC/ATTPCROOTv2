104 Int_t AtPulseLineTask::integrateTimebuckets(std::vector<double> &zIntegral, AtSimulatedLine *line)
105 {
106  zIntegral.clear();
107 
108  // Shift the times by the pad plane location
109  auto tMax = line->GetInitialPosition().z() + fTBPadPlane * fTBTime;
110  auto tMin = line->GetFinalPosition().z() + fTBPadPlane * fTBTime;
111  auto dT = (tMax - tMin);
112  if (dT > fTBTime) {
113  LOG(warn) << "This line charge spans multiple TB widths from " << tMin << " us to " << tMax << " us.";
114  LOG(warn) << " The time distribution is likely widder than will be calculated. dT: " << dT
115  << "ns TB width: " << fTBTime << "ns";
116  }
117 
118  auto tMean = (tMax + tMin) / 2.;
119  auto tIntegrationMinimum = tMin - fNumSigmaToIntegrateZ * line->GetLongitudinalDiffusion();
120  auto tIntegrationMaximum = tMax + fNumSigmaToIntegrateZ * line->GetLongitudinalDiffusion();
121 
122  const TAxis *axis = eleAccumulated[0]->GetXaxis();
123  auto binMin = axis->FindBin(tIntegrationMinimum);
124  auto binMax = axis->FindBin(tIntegrationMaximum);
125  if (binMin < fTBPadPlane)
126  binMin = fTBPadPlane;
127  if (binMax > fTBEntrance)
128  binMax = fTBEntrance;
129 
130  // Integrate G(tMean, sigmaLongDiff) over each time bucket from binMin to binMax
131  // and fill zIntegral with the result.
132  auto denominator = line->GetLongitudinalDiffusion() * TMath::Sqrt(2);
133  double lowerBound = TMath::Erf((axis->GetBinLowEdge(binMin) - tMean) / denominator);
134  for (int i = binMin; i <= binMax; ++i) {
135  auto upperBound = TMath::Erf((axis->GetBinUpEdge(i) - tMean) / denominator);
136  auto integral = 0.5 * (upperBound - lowerBound);
137  zIntegral.emplace_back(integral);
138  lowerBound = upperBound;
139  }
140 
141  // Renormalize the integral if we're up against the pad plane or window
142  if (binMin == fTBPadPlane || binMax == fTBEntrance) {
143  auto sum = std::accumulate(zIntegral.begin(), zIntegral.end(), 0.);
144  for (auto &elem : zIntegral)
145  elem /= sum;
146  }
147 
148  return binMin;
149 }

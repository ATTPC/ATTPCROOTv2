120 void AtPulseTask::saveMCInfo(int mcPointID, int padNumber, int trackID)
121 {
122  // Count occurrences of simPoints coming from the same mcPoint
123  int count = 0;
124 
125  // The same MC point ID is saved per pad only once, but duplicates are allowed in other pads
126  for (auto it = MCPointsMap.lower_bound(padNumber); it != MCPointsMap.upper_bound(padNumber); ++it) {
127  auto mcPointMap = dynamic_cast<AtMCPoint *>(fMCPointArray->At(mcPointID));
128  auto trackIDMap = mcPointMap->GetTrackID();
129  if (it->second == mcPointID || (trackID == trackIDMap)) {
130  ++count;
131  break;
132  }
133  }
134 
135  // insert if the mcPointID and trackID do not both match any existing point
136  if (count == 0)
137  MCPointsMap.insert(std::make_pair(padNumber, mcPointID));
138 }

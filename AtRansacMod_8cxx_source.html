639 Double_t AtRansacMod::Fit3D(vector<int> inliners, TVector3 &V1, TVector3 &V2)
640 {
641 
642  //------3D Line Regression
643  //----- adapted from: http://fr.scribd.com/doc/31477970/Regressions-et-trajectoires-3D
644  int R, C;
645  double Q;
646  double Xm, Ym, Zm;
647  double Xh, Yh, Zh;
648  double a, b;
649  double Sxx, Sxy, Syy, Sxz, Szz, Syz;
650  double theta;
651  double K11, K22, K12, K10, K01, K00;
652  double c0, c1, c2;
653  double p, q, r, dm2;
654  double rho, phi;
655 
656  Q = Xm = Ym = Zm = 0.;
657  double total_charge = 0;
658  Sxx = Syy = Szz = Sxy = Sxz = Syz = 0.;
659 
660  for (auto i : inliners) {
661  Q += vQ[i] / 10.;
662  Xm += vX[i] * vQ[i] / 10.;
663  Ym += vY[i] * vQ[i] / 10.;
664  Zm += vZ[i] * vQ[i] / 10.;
665  Sxx += vX[i] * vX[i] * vQ[i] / 10.;
666  Syy += vY[i] * vY[i] * vQ[i] / 10.;
667  Szz += vZ[i] * vZ[i] * vQ[i] / 10.;
668  Sxy += vX[i] * vY[i] * vQ[i] / 10.;
669  Sxz += vX[i] * vZ[i] * vQ[i] / 10.;
670  Syz += vY[i] * vZ[i] * vQ[i] / 10.;
671  }
672  // vTrackCharge.push_back(total_charge);
673 
674  Xm /= Q;
675  Ym /= Q;
676  Zm /= Q;
677  Sxx /= Q;
678  Syy /= Q;
679  Szz /= Q;
680  Sxy /= Q;
681  Sxz /= Q;
682  Syz /= Q;
683  Sxx -= (Xm * Xm);
684  Syy -= (Ym * Ym);
685  Szz -= (Zm * Zm);
686  Sxy -= (Xm * Ym);
687  Sxz -= (Xm * Zm);
688  Syz -= (Ym * Zm);
689 
690  theta = 0.5 * atan((2. * Sxy) / (Sxx - Syy));
691 
692  K11 = (Syy + Szz) * pow(cos(theta), 2) + (Sxx + Szz) * pow(sin(theta), 2) - 2. * Sxy * cos(theta) * sin(theta);
693  K22 = (Syy + Szz) * pow(sin(theta), 2) + (Sxx + Szz) * pow(cos(theta), 2) + 2. * Sxy * cos(theta) * sin(theta);
694  // K12 = -Sxy * (pow(cos(theta), 2) - pow(sin(theta), 2)) + (Sxx - Syy) * cos(theta) * sin(theta);
695  K10 = Sxz * cos(theta) + Syz * sin(theta);
696  K01 = -Sxz * sin(theta) + Syz * cos(theta);
697  K00 = Sxx + Syy;
698 
699  c2 = -K00 - K11 - K22;
700  c1 = K00 * K11 + K00 * K22 + K11 * K22 - K01 * K01 - K10 * K10;
701  c0 = K01 * K01 * K11 + K10 * K10 * K22 - K00 * K11 * K22;
702 
703  p = c1 - pow(c2, 2) / 3.;
704  q = 2. * pow(c2, 3) / 27. - c1 * c2 / 3. + c0;
705  r = pow(q / 2., 2) + pow(p, 3) / 27.;
706 
707  if (r > 0)
708  dm2 = -c2 / 3. + pow(-q / 2. + sqrt(r), 1. / 3.) + pow(-q / 2. - sqrt(r), 1. / 3.);
709  else {
710  rho = sqrt(-pow(p, 3) / 27.);
711  phi = acos(-q / (2. * rho));
712  dm2 = min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos(phi / 3.),
713  min(-c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 2. * TMath::Pi()) / 3.),
714  -c2 / 3. + 2. * pow(rho, 1. / 3.) * cos((phi + 4. * TMath::Pi()) / 3.)));
715  }
716 
717  a = -K10 * cos(theta) / (K11 - dm2) + K01 * sin(theta) / (K22 - dm2);
718  b = -K10 * sin(theta) / (K11 - dm2) - K01 * cos(theta) / (K22 - dm2);
719 
720  Xh = ((1. + b * b) * Xm - a * b * Ym + a * Zm) / (1. + a * a + b * b);
721  Yh = ((1. + a * a) * Ym - a * b * Xm + b * Zm) / (1. + a * a + b * b);
722  Zh = ((a * a + b * b) * Zm + a * Xm + b * Ym) / (1. + a * a + b * b);
723 
724  V1.SetXYZ(Xm, Ym, Zm);
725  V2.SetXYZ(Xh, Yh, Zh);
726 
727  return (fabs(dm2 / Q));
728 }
